    <script>
        // Раскрытие/скрытие статей
        function toggleArticle(id) {
            const article = document.getElementById(`article-${id}`);
            const isVisible = article.style.display === 'block';
            
            // Сначала скрываем все статьи
            document.querySelectorAll('.full-article').forEach(el => {
                el.style.display = 'none';
            });
            
            // Показываем/скрываем выбранную
            article.style.display = isVisible ? 'none' : 'block';
            
            // Плавная прокрутка к статье
            if (!isVisible) {
                article.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
            }
        }

        // Плавное появление статей при прокрутке
        const fadeElements = document.querySelectorAll('.fade-in');
        
        function checkFade() {
            fadeElements.forEach(element => {
                const elementTop = element.getBoundingClientRect().top;
                const windowHeight = window.innerHeight;
                
                if (elementTop < windowHeight - 100) {
                    element.style.opacity = 1;
                    element.style.transform = 'translateY(0)';
                }
            });
        }
        
        window.addEventListener('scroll', checkFade);
        window.addEventListener('load', checkFade);

        // Закрытие статей при клике на изображение
        document.querySelectorAll('.post-image').forEach(image => {
            image.addEventListener('click', function() {
                const articleId = this.closest('.blog-post').querySelector('.full-article').id;
                const id = articleId.split('-')[1];
                toggleArticle(id);
            });
        });
    </script>
